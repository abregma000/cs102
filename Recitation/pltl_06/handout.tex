\section*{Swap}
Write a function that uses the following prototype and swaps the values of the two reference parameters.
\begin{lstlisting}
	void swap(int &a, int &b);
\end{lstlisting}
Use the following main to test your implementation:
\begin{lstlisting}{c++}
#include<iostream>
using namespace std;
void swap(int &a, int &b);

int main(){
	int a, b;
	cout<<"Enter a: ";
	cint>>a;
	cout<<"Enter b: ";
	cin>>b;
	int old_a = a;
	int old_b = b;
	swap(a, b);
	if ((old_a==b) && (old_b==a)){
		cout<<"Correct Implementation"endl;
	}else{
		cout<<"Incorrect Implementation"<<endl;
	}
return 0;
}
\end{lstlisting}

\section*{Sort}

\begin{lstlisting}{c++}

\end{lstlisting}
Use the following main to test:
\begin{lstlisting}{c++}

using namespace std;
void sort(int &a, int &b, int &c);

int main(){

	}
